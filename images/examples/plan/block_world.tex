\begin{document}
	\newcommand{\block}[3]{
		\node[ell, rectangle, minimum height = 15, minimum width = 15] (#3) at (#1 pt,#2 pt) {#3}; 
	}
	\newcommand{\hand}[2]{
		\draw[ell] (#1 pt, #2 pt) -- ++(-10 pt, -7 pt);
		\draw[ell] (#1 pt, #2 pt) -- ++(10 pt, -7 pt);
		\draw[ell] (#1-10 pt, #2-7 pt) -- ++(5 pt, -7 pt);
		\draw[ell] (#1+10 pt, #2-7 pt) -- ++(-5 pt, -7 pt);
		
		\draw[ell] (#1 pt, #2 pt) -- ++(0 pt, 7 pt);
		\draw[ell] (#1 pt, #2 + 7 pt) -- ++(10 pt, 0 pt);
	}
	\newcommand{\handp}[2]{
		\draw[ell] (#1 pt, #2 pt) -- ++(-10 pt, -7 pt);
		\draw[ell] (#1 pt, #2 pt) -- ++(10 pt, -7 pt);
		\draw[ell] (#1-10 pt, #2-7 pt) -- ++(2 pt, -7 pt);
		\draw[ell] (#1+10 pt, #2-7 pt) -- ++(-2 pt, -7 pt);
		
		\draw[ell] (#1 pt, #2 pt) -- ++(0 pt, 7 pt);
		\draw[ell] (#1 pt, #2 + 7 pt) -- ++(10 pt, 0 pt);
	}
