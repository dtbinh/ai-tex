	\begin{abstract}
		\noindent Рассматривается семантический уровень описания функций, которые в психологии принято относить к функциям сознания и самосознания. Исследуется механизм работы компонент знака, введённых в первой части статьи. С использованием введённого описания семантического уровня исследуется сходимость основного итерационного процесса образования знака "--- связывания образной компоненты знака и его значения. Введение алгоритмов работы компонент знака позволяет построить алгоритм процесса синтеза плана поведения, а также построить новую архитектуру интеллектуальных агентов, обладающих, в частности, способностями к распределению ролей в коалициях.
	\end{abstract}

	\section{Алгоритм планирования}	Планом $Plan$ будем называть такую последовательность пар <<ситуация "--- действие>> , в которой.
	
	Целевая ситуация строится исходя из образа действия, связанного с личностным смыслом, который был определён в процессе целеполагания для целевого знака.
	\begin{algorithm}
		\caption{Алгоритм $\mathfrak{A}_{beh}$}\label{alg:beh_plan}
		\begin{algorithmic}[1]
			\Require начальная ситуация $F_{sit}$, целевая ситуация $F_{goal}$, функция оценки $\Phi_a$;
			\Ensure план $Plan$;
			\algrule
			\State $Plan=\varnothing; i = 0$;
			\State $\Delta^{(i)}=F_{sit}\setminus F_{goal}$; \Comment{текущая невязка состояний}
			\State $F_{goal}^{(i)} = F_{goal}$; \Comment{текущая подцель}
			
			\While $\Delta^{(i)}\not=\varnothing$ 
				\State $F_p^{(i)}=\{f:f = \argmin\limits_{f_p\in F_{sit}} (|F_A(f_p)\setminus\Delta^{(i)}|)\}$; \Comment{находим множество наиболее подходящих действий}
				\State $F_a^{(i)} = \varnothing$; \Comment{текущее множестов личностных смыслов}
				\ForAll $f_p\in F_p^{(i)}$
					\State $F_a^{(i)} = F_a^{(i)}\cup \{\Call{Interior}{f_p}\}$;\Comment{интериоризация значения}
				\EndFor
				\State $f_a=\Phi_a(F_a^{(i)},f_{goal})$; \Comment{выбор предпочитаемого личностного смысла}
				\If{$F_C(f_a)\subseteq F_{sit}$}
					\State \Return $Plan$;		\Comment{построена цепочка применимых действий "--- план построен}
				\Else
					\State $Plan = \{f_a\}\cup Plan$; \Comment{добавляем текущее действие в план}
				\EndIf
			\EndWhile
		\end{algorithmic}
	\end{algorithm}
